1. 载入网页，进入调试，输入下列代码
``` js
var allText = document.body.innerText;
var textLength = allText.length;
var i = 0;
var interval = setInterval(function() {
    if (i == textLength) clearInterval(interval);
    else {
        console.log(allText.charAt(i));
        i++;
    }
}, 100);

```
